#include int calculateLength(char*); void main() { char str1[25]; int l; printf(" Input a string : "); scanf("%s",str1); l = calculateLength(str1); printf(" The length of the given string %s is : %d ", str1, l); } int calculateLength(char* ch) { int ctr = 0; while (*ch != '\0') { ctr++; ch++; } return ctr; }
